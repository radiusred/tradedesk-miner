    ```
    use clap::{Parser, Subcommand};
    use std::path::{Path, PathBuf};
    use miner_core::config::{CliOverrides, OutputDest};

    /// tradedesk-miner CLI
    #[derive(Debug, Parser)]
    #[command(name = "miner", version, about)]
    pub struct Cli {
        /// Explicit config file path (overrides XDG / CWD lookup)
        #[arg(long, global = true)]
        pub config: Option<PathBuf>,

        /// Override the source cache root (CLI > env > TOML)
        #[arg(long, global = true, env = "MINER_CACHE_ROOT")]
        pub cache_root: Option<PathBuf>,

        /// Override the derived-bar cache root
        #[arg(long, global = true, env = "MINER_BAR_CACHE_ROOT")]
        pub bar_cache_root: Option<PathBuf>,

        /// Override the output destination
        #[arg(long, global = true, env = "MINER_OUTPUT")]
        pub output: Option<String>,    // parsed to OutputDest below; clap doesn't directly grok the enum

        #[command(subcommand)]
        pub command: Command,
    }

    #[derive(Debug, Subcommand)]
    pub enum Command {
        /// Emit one RunStart + one RunEnd fixture to stdout (Phase 1 sanity check; no scans yet)
        EmitFixture,
    }

    impl Cli {
        /// Convert clap-parsed values to the figment overlay struct.
        pub fn overrides(&self) -> CliOverrides {
            CliOverrides {
                cache_root: self.cache_root.clone(),
                bar_cache_root: self.bar_cache_root.clone(),
                output: self.output.as_deref().map(|s| match s {
                    "stdout" => OutputDest::Stdout,
                    other => OutputDest::File(PathBuf::from(other)),
                }),
            }
        }
    }

    /// Resolve the TOML config path: --config explicit > XDG/platform-native > CWD > None.
    ///
    /// Cross-platform note (see PLAN must_haves): on Linux this resolves to `$XDG_CONFIG_HOME/miner/miner.toml`
    /// (or `~/.config/miner/miner.toml`) per D-16. On macOS it resolves to
    /// `~/Library/Application Support/miner/miner.toml` and on Windows to
    /// `%APPDATA%/miner/miner.toml` — platform-native, NOT XDG. v1 is Linux-first; the divergence
    /// is deliberate and documented.
    pub fn resolve_toml_path(cli_explicit: Option<&Path>) -> Option<PathBuf> {
        if let Some(p) = cli_explicit { return Some(p.to_path_buf()); }
        if let Some(proj) = directories::ProjectDirs::from("", "", "miner") {
            let xdg = proj.config_dir().join("miner.toml");
            if xdg.exists() { return Some(xdg); }
        }
        let cwd = Path::new("./miner.toml");
        if cwd.exists() { return Some(cwd.to_path_buf()); }
        None
    }
    ```

    ```
    /// parse errors / Message variants) → InvalidConfig. Mapping every error to
    /// MissingRequiredConfig is a SEMANTIC BUG per the planner's PLAN 05 review.
    fn classify_figment_error(err: &figment::Error) -> PreflightCode {
        use figment::error::Kind;
        // figment::Error is iterable over potentially multiple inner errors;
        // we classify by the FIRST one (the proximate cause).
        let first_kind = err.clone().into_iter().next().map(|e| e.kind).unwrap_or(Kind::Message(String::new()));
        match first_kind {
            Kind::MissingField(_) => PreflightCode::MissingRequiredConfig,
            // Type / value mismatches (TOML had wrong type for a field, env var unparseable, etc.)
            Kind::InvalidType(_, _) | Kind::InvalidValue(_, _) | Kind::InvalidLength(_, _) => PreflightCode::InvalidConfig,
            // Parse errors, malformed TOML, unrecognised fields, custom messages
            Kind::Message(_) | Kind::UnknownVariant(_, _) | Kind::UnknownField(_, _) | Kind::DuplicateField(_) => PreflightCode::InvalidConfig,
            // Catch-all: any future Kind variant we haven't enumerated → safest is InvalidConfig
            // (NOT MissingRequiredConfig — that would falsely claim the user forgot a field).
            _ => PreflightCode::InvalidConfig,
        }
    }

    fn emit_fixture() -> anyhow::Result<()> {
        tracing::info!("emitting fixture");
        let mut sink = StdoutSink::new();
        let run_id = RunId::new();
        let started = chrono::Utc::now();

        let start = Finding::RunStart(RunStart {
            run_id,                  // RunId is Copy (Plan 03 acceptance criterion) — moved here
            started_at_utc: started,
            miner_version: env!("CARGO_PKG_VERSION").to_string(),
            code_revision: miner_core::CODE_REVISION.to_string(),

        });

    }
    ```
