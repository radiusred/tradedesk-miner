    ```
    //! xtask: dev-only commands for tradedesk-miner.
    //!
    //! xtask is dev-only and never produces findings, so it is exempted from the
    //! workspace stdout discipline (clippy::disallowed_macros).
    //!
    //! Per RESEARCH §Schema Derivation Strategy ("lint scoping nuance").
    #![allow(clippy::disallowed_macros)]

    use clap::{Parser, Subcommand};
    use miner_core::Finding;
    use std::path::PathBuf;

    #[derive(Parser)]
    #[command(name = "xtask", about = "dev-only tasks for tradedesk-miner")]
    struct Cli {
        #[command(subcommand)]
        cmd: Cmd,
    }

    #[derive(Subcommand)]
    enum Cmd {
        /// Regenerate `schemas/findings-v1.schema.json` from miner-core types.
        GenSchema {
            #[arg(default_value = "schemas/findings-v1.schema.json")]
            out: PathBuf,
        },
    }

    fn main() -> anyhow::Result<()> {
        let cli = Cli::parse();
        match cli.cmd {
            Cmd::GenSchema { out } => gen_schema(out),
        }
    }

    fn gen_schema(out: PathBuf) -> anyhow::Result<()> {
        if let Some(parent) = out.parent() {
            std::fs::create_dir_all(parent)?;
        }
        // Determinism pipeline (see PLAN must_haves):
        // 1. schemars produces a Schema (schemars 1.x; stable derive walk order)
        // 2. round-trip through serde_json::Value so the map is BTreeMap-backed
        //    (workspace pins serde_json without `preserve_order` per Plan 01-01)
        // 3. to_string_pretty emits keys in BTreeMap order — byte-stable
        let schema = schemars::schema_for!(Finding);
        let as_value: serde_json::Value = serde_json::to_value(&schema)?;
        let pretty = serde_json::to_string_pretty(&as_value)?;
        std::fs::write(&out, format!("{}\n", pretty))?;
        eprintln!("wrote {}", out.display());
        Ok(())
    }
    ```

    Note: the `#![allow(clippy::disallowed_macros)]` is correctly scoped to the whole `xtask` crate per RESEARCH's "lint scoping nuance" — xtask is dev-only and the `eprintln!` is for developer feedback, not a finding.

    The `serde_json::to_value` → `to_string_pretty` normalisation step is what guarantees byte-stability across multiple `gen-schema` invocations, even if schemars' internal `Map` happens to be insertion-ordered. The intermediate `serde_json::Value` is BTreeMap-backed (because Plan 01-01 deliberately did NOT enable `serde_json/preserve_order`), so the final pretty-print emits keys alphabetically.
