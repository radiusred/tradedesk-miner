    fn mask_volatile_fields(v: &mut serde_json::Value) {
        if let serde_json::Value::Object(map) = v {
            for key in ["run_id", "started_at_utc", "ended_at_utc"] {
                if map.contains_key(key) {
                    map.insert(key.to_string(), serde_json::Value::String(format!("<masked_{key}>")));
                }
            }
            if map.contains_key("wall_clock_ms") {
                map.insert("wall_clock_ms".to_string(), serde_json::Value::from(0));
            }
            // Recurse into nested objects (the `request` echo on RunStart is the only known nested
            // structure but recursion is cheap insurance against future shape changes).
            for (_, child) in map.iter_mut() {
                mask_volatile_fields(child);
            }
        } else if let serde_json::Value::Array(arr) = v {
            for child in arr.iter_mut() { mask_volatile_fields(child); }
        }
    }

    fn mask_emit_fixture_stdout(raw: &str) -> Vec<String> {
        raw.lines()
            .filter(|l| !l.is_empty())
            .map(|line| {
                let mut v: serde_json::Value = serde_json::from_str(line).expect("emit-fixture line is valid JSON");
                mask_volatile_fields(&mut v);
                serde_json::to_string(&v).expect("masked JSON re-serialises")
            })
            .collect()
    }

    #[test]

    fn emit_fixture_byte_identical_when_volatile_fields_masked() {

    ## Quickstart

    1. **Prerequisites:** Rust 1.85+ stable (`rustup default 1.85`), git.

    2. **Clone and build:**
       ```
       git clone https://github.com/radiusred/tradedesk-miner
       cd tradedesk-miner
       cargo build --workspace
       ```

    3. **Emit a fixture (the Phase 1 smoke test):**
       ```
       MINER_CACHE_ROOT=/tmp/cache \
       MINER_BAR_CACHE_ROOT=/tmp/bar \
       MINER_OUTPUT=stdout \
       cargo run -p miner-cli -- emit-fixture
       ```

       Expected output on stdout: 2 lines of NDJSON — one `run_start`, one `run_end`. Tracing log lines on stderr.

    4. **Inspect the locked envelope schema:**
       ```
       cat schemas/findings-v1.schema.json | jq '.title, ."$defs" | keys?'
       ```

    5. **Run the test suite:**
       ```
       cargo test --workspace
       cargo clippy --workspace --all-targets -- -D warnings
       ```

| Test subprocess → CI runner | The `cli_streams` integration tests spawn the actual `miner` binary, which means the tests catch any wrapper-level bug that the unit tests miss (e.g., tracing-subscriber not initialised before clap parsing leaks into stdout). |
| Schema artifact ↔ test instance | The `schema_roundtrip` test loads the COMMITTED `findings-v1.schema.json` (not a freshly-generated one) and validates against it. If a contributor changes Rust types without regenerating the schema, this test fails immediately. |
